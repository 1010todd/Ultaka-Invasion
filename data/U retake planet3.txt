mission "UiMW: Player Conquer New Britain"
	landing
	invisible
	source "New Britain"
	name "Conquer New Britain"
	description `Retake New Britain`
	to offer
		has "event: UiMW: Ultaka conquer 3"
		has "tribute: New Britain"
	on offer
		conversation
			`You've chased away the Ultaka stationed here but to hold it you'll need a fleet to defend it, check the job board to see what fleet you can buy.`
				accept
		event "UiMW: Player conquer New Britain"
		clear "UiMW: Ultaka Reconquer New Britain notice: offered"
	on accept
		fail
		
mission "UiMW: Ultaka Reconquer New Britain notice"
	landing
	repeat
	source
		not planet "New Britain"
	to offer
		"uimw: defense counter: new britain" <= 0
		not "UiMW: Ultaka Reconquer New Britain notice: offered"
		has "UiMW: Player Conquer New Britain: offered"
	on offer
		conversation
			`NOTICE: Your fleet defending New Britain have been defeated and the planet was taken over by the Ultaka.`
			`	[note: map will update after departing. Relinquish Tribute and Demand Tribute again to reconquer it.]`
				decline
		event "UiMW: Ultaka reconquer New Britain"
		clear "UiMW: Player Conquer New Britain: offered"
		clear "tribute: New Britain"
				
mission "UiMW: Ultaka Reconquer New Britain warning"
	landing
	repeat
	source "New Britain"
	to offer
		"uimw: defense counter: new britain" <= 0
		has "UiMW: Player Conquer New Britain: offered"
	on offer
		conversation
			`WARN: Your do not have any active fleet defending <origin>, Ultaka will come back and take it if you leave it undefended.`
				decline

mission "UiMW: New Britain Defense [Militia]"
	job
	repeat
	source "New Britain"
	name `New Britain Defense (Militia)`
	description `Buy very large militia fleet consist of ships from Southbound and Tarazed shipyards to defend New Britain for <payment>.`
	to offer
		#"net worth" >= 100000000
		has "tribute: New Britain"
	on accept
		payment -75000000
		"uimw: defense counter: new britain" += 1
	# Used kill to require all ships to be dead, save requires only 1 ship dead to fail.
	npc kill
		government "Escort"
		personality heroic uninterested staying
		fleet
			names "player navy"
			fighters "player fighters"
			variant
				"Falcon (Heavy)" 5
				"Falcon" 5
				"Bastion (Heavy)" 7
				"Argosy (Missile)" 12
				"Argosy (Turret)" 12
				"Osprey (Missile)" 7
				"Fury" 12
		conversation #Required because on complete have specific destination and save don't work, so conversation is used instead to trigger the value reduction.
			`Your New Britain militia defense fleet has been wiped out!`
			action
				"uimw: defense counter: new britain" -= 1
	on abort
		payment 75000000
		"uimw: defense counter: new britain" -= 1
